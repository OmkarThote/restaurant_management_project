from bs4 import BeautifulSoup # Sample HTML content html_content = """ Accessibility Example

Welcome to My Website

""" # Parse the HTML soup = BeautifulSoup(html_content, "html.parser") # Add descriptive alt text to images without alt attribute for img in soup.find_all("img"): # You can customize the alt text based on filename or context img['alt'] = f"Image of {img['src'].split('.')[0]}" # Print updated HTML print(soup.prettify())
